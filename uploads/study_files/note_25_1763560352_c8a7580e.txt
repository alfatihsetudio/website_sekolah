<?php
// space_belajar/file_exploler.php
require_once __DIR__ . '/../inc/auth.php';
require_once __DIR__ . '/../inc/db.php';
require_once __DIR__ . '/../inc/helpers.php';

requireLogin();

$role = getUserRole();
if (!in_array($role, ['murid','siswa'], true)) {
    http_response_code(403);
    echo "Halaman ini khusus untuk akun murid.";
    exit;
}

$db      = getDB();
$baseUrl = rtrim(BASE_URL, '/\\');
$userId  = (int)($_SESSION['user_id'] ?? 0);

if ($userId <= 0) {
    http_response_code(403);
    echo "Sesi login tidak valid.";
    exit;
}

// Folder upload
$uploadBaseDir = __DIR__ . '/../uploads/study_files';
if (!is_dir($uploadBaseDir)) {
    @mkdir($uploadBaseDir, 0777, true);
}

function sb_clean($v) {
    return trim((string)$v);
}

// Folder aktif
$currentFolderId = isset($_GET['folder']) ? (int)$_GET['folder'] : 0;

// Pastikan folder aktif milik user
if ($currentFolderId > 0) {
    $stmt = $db->prepare("SELECT id FROM study_folders WHERE id = ? AND user_id = ? LIMIT 1");
    if ($stmt) {
        $stmt->bind_param("ii", $currentFolderId, $userId);
        $stmt->execute();
        $res = $stmt->get_result();
        if (!$res || !$res->num_rows) {
            $currentFolderId = 0;
        }
        $stmt->close();
    } else {
        $currentFolderId = 0;
    }
}

// Search
$searchQuery = sb_clean($_GET['q'] ?? '');

// File untuk preview & edit catatan
$viewFileId = isset($_GET['view']) ? (int)$_GET['view'] : 0;
$editFileId = isset($_GET['edit']) ? (int)$_GET['edit'] : 0;

$viewFile    = null;
$editFile    = null;
$editContent = '';

// Ambil data file untuk preview
if ($viewFileId > 0) {
    $stmt = $db->prepare("
        SELECT id, user_id, folder_id, title, original_name, stored_name, mime_type, size_bytes, description, created_at
        FROM study_files
        WHERE id = ? AND user_id = ?
        LIMIT 1
    ");
    if ($stmt) {
        $stmt->bind_param("ii", $viewFileId, $userId);
        $stmt->execute();
        $res = $stmt->get_result();
        if ($res && $res->num_rows) {
            $viewFile = $res->fetch_assoc();
        }
        $stmt->close();
    }
}

// Ambil data file untuk edit catatan (jika ada)
if ($editFileId > 0) {
    $stmt = $db->prepare("
        SELECT id, user_id, folder_id, title, original_name, stored_name, mime_type, size_bytes, description, created_at
        FROM study_files
        WHERE id = ? AND user_id = ?
        LIMIT 1
    ");
    if ($stmt) {
        $stmt->bind_param("ii", $editFileId, $userId);
        $stmt->execute();
        $res = $stmt->get_result();
        if ($res && $res->num_rows) {
            $editFile = $res->fetch_assoc();
            if (!empty($editFile['stored_name'])) {
                $fsPath = $uploadBaseDir . '/' . $editFile['stored_name'];
                if (is_file($fsPath) && is_readable($fsPath)) {
                    $editContent = (string)file_get_contents($fsPath);
                }
            }
        }
        $stmt->close();
    }
}

// -------------------- HANDLE POST -------------------- //
$errors     = [];
$successMsg = '';

if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $action = $_POST['action'] ?? '';

    // 1) Buat folder
    if ($action === 'create_folder') {
        $folderName = sb_clean($_POST['folder_name'] ?? '');
        $parentId   = (int)($_POST['parent_id'] ?? 0);

        if ($folderName === '') {
            $errors[] = 'Nama folder tidak boleh kosong.';
        } else {
            if ($parentId > 0) {
                $stmt = $db->prepare("SELECT id FROM study_folders WHERE id = ? AND user_id = ? LIMIT 1");
                if ($stmt) {
                    $stmt->bind_param("ii", $parentId, $userId);
                    $stmt->execute();
                    $r = $stmt->get_result();
                    if (!$r || !$r->num_rows) {
                        $parentId = 0;
                    }
                    $stmt->close();
                } else {
                    $parentId = 0;
                }
            }

            $stmt = $db->prepare("
                INSERT INTO study_folders (user_id, parent_id, name, created_at, updated_at)
                VALUES (?, ?, ?, NOW(), NOW())
            ");
            if ($stmt) {
                $stmt->bind_param("iis", $userId, $parentId, $folderName);
                if ($stmt->execute()) {
                    header('Location: ' . $baseUrl . '/space_belajar/file_exploler.php?folder=' . $parentId);
                    exit;
                } else {
                    $errors[] = 'Gagal membuat folder: ' . $stmt->error;
                }
                $stmt->close();
            } else {
                $errors[] = 'Gagal menyiapkan query folder.';
            }
        }

    // 2) Upload file (termasuk kamera)
    } elseif ($action === 'upload_file') {
        $parentId = (int)($_POST['parent_id'] ?? 0);

        if ($parentId > 0) {
            $stmt = $db->prepare("SELECT id FROM study_folders WHERE id = ? AND user_id = ? LIMIT 1");
            if ($stmt) {
                $stmt->bind_param("ii", $parentId, $userId);
                $stmt->execute();
                $r = $stmt->get_result();
                if (!$r || !$r->num_rows) {
                    $parentId = 0;
                }
                $stmt->close();
            }
        }

        if (empty($_FILES['upload_file']) || $_FILES['upload_file']['error'] !== UPLOAD_ERR_OK) {
            $errors[] = 'Pilih file yang ingin diunggah.';
        } else {
            $tmp      = $_FILES['upload_file']['tmp_name'];
            $origName = $_FILES['upload_file']['name'];
            $size     = (int)$_FILES['upload_file']['size'];
            $mime     = $_FILES['upload_file']['type'] ?? 'application/octet-stream';

            if ($size <= 0) {
                $errors[] = 'Ukuran file tidak valid.';
            } else {
                $ext       = pathinfo($origName, PATHINFO_EXTENSION);
                $randomKey = bin2hex(random_bytes(4));
                $stored    = time() . '_' . $randomKey . ($ext ? '.' . $ext : '');
                $destPath  = $uploadBaseDir . '/' . $stored;

                if (!@move_uploaded_file($tmp, $destPath)) {
                    $errors[] = 'Gagal menyimpan file di server.';
                } else {
                    $title = sb_clean($_POST['file_title'] ?? '');
                    if ($title === '') {
                        $title = $origName;
                    }
                    $desc      = sb_clean($_POST['file_desc'] ?? '');
                    $sizeBytes = $size;

                    $stmt = $db->prepare("
                        INSERT INTO study_files (user_id, folder_id, title, original_name, stored_name, mime_type, size_bytes, description, created_at)
                        VALUES (?, ?, ?, ?, ?, ?, ?, ?, NOW())
                    ");
                    if ($stmt) {
                        $stmt->bind_param("iissssis", $userId, $parentId, $title, $origName, $stored, $mime, $sizeBytes, $desc);
                        if ($stmt->execute()) {
                            header('Location: ' . $baseUrl . '/space_belajar/file_exploler.php?folder=' . $parentId);
                            exit;
                        } else {
                            $errors[] = 'Gagal menyimpan data file: ' . $stmt->error;
                        }
                        $stmt->close();
                    } else {
                        $errors[] = 'Gagal menyiapkan query file.';
                    }
                }
            }
        }

    // 3) Tambah catatan
    } elseif ($action === 'add_note') {
        $parentId    = (int)($_POST['parent_id'] ?? 0);
        $noteTitle   = sb_clean($_POST['note_title'] ?? '');
        $noteContent = (string)($_POST['note_content'] ?? '');

        if ($noteTitle === '') {
            $errors[] = 'Judul catatan tidak boleh kosong.';
        } elseif ($noteContent === '') {
            $errors[] = 'Isi catatan tidak boleh kosong.';
        } else {
            if ($parentId > 0) {
                $stmt = $db->prepare("SELECT id FROM study_folders WHERE id = ? AND user_id = ? LIMIT 1");
                if ($stmt) {
                    $stmt->bind_param("ii", $parentId, $userId);
                    $stmt->execute();
                    $r = $stmt->get_result();
                    if (!$r || !$r->num_rows) {
                        $parentId = 0;
                    }
                    $stmt->close();
                }
            }

            $randomKey = bin2hex(random_bytes(4));
            $stored    = 'note_' . $userId . '_' . time() . '_' . $randomKey . '.txt';
            $destPath  = $uploadBaseDir . '/' . $stored;

            if (file_put_contents($destPath, $noteContent) === false) {
                $errors[] = 'Gagal menyimpan catatan ke file.';
            } else {
                $origName  = $stored;
                $mime      = 'text/plain';
                $sizeBytes = filesize($destPath);

                $stmt = $db->prepare("
                    INSERT INTO study_files (user_id, folder_id, title, original_name, stored_name, mime_type, size_bytes, description, created_at)
                    VALUES (?, ?, ?, ?, ?, ?, ?, '', NOW())
                ");
                if ($stmt) {
                    $stmt->bind_param("iissssi", $userId, $parentId, $noteTitle, $origName, $stored, $mime, $sizeBytes);
                    if ($stmt->execute()) {
                        header('Location: ' . $baseUrl . '/space_belajar/file_exploler.php?folder=' . $parentId);
                        exit;
                    } else {
                        $errors[] = 'Gagal menyimpan catatan ke database: ' . $stmt->error;
                    }
                    $stmt->close();
                } else {
                    $errors[] = 'Gagal menyiapkan query catatan.';
                }
            }
        }

    // 4) Update catatan
    } elseif ($action === 'update_note') {
        $fileId      = (int)($_POST['file_id'] ?? 0);
        $noteTitle   = sb_clean($_POST['note_title'] ?? '');
        $noteContent = (string)($_POST['note_content'] ?? '');

        if ($fileId <= 0) {
            $errors[] = 'Catatan tidak ditemukan.';
        } else {
            $stmt = $db->prepare("
                SELECT id, folder_id, stored_name, mime_type
                FROM study_files
                WHERE id = ? AND user_id = ?
                LIMIT 1
            ");
            if ($stmt) {
                $stmt->bind_param("ii", $fileId, $userId);
                $stmt->execute();
                $res = $stmt->get_result();
                if ($res && $res->num_rows) {
                    $row       = $res->fetch_assoc();
                    $folderId  = (int)$row['folder_id'];
                    $stored    = $row['stored_name'];
                    $mime      = $row['mime_type'];
                    $stmt->close();

                    $path = $uploadBaseDir . '/' . $stored;
                    if ($mime === 'text/plain' && is_file($path) && is_writable($path)) {
                        if (file_put_contents($path, $noteContent) === false) {
                            $errors[] = 'Gagal memperbarui isi catatan.';
                        } else {
                            $newSize = filesize($path);
                            $stmtU = $db->prepare("
                                UPDATE study_files
                                SET title = ?, size_bytes = ?
                                WHERE id = ? AND user_id = ?
                            ");
                            if ($stmtU) {
                                $stmtU->bind_param("siii", $noteTitle, $newSize, $fileId, $userId);
                                if ($stmtU->execute()) {
                                    header('Location: ' . $baseUrl . '/space_belajar/file_exploler.php?folder=' . $folderId . '&view=' . $fileId);
                                    exit;
                                } else {
                                    $errors[] = 'Gagal update data catatan: ' . $stmtU->error;
                                }
                                $stmtU->close();
                            } else {
                                $errors[] = 'Gagal menyiapkan query update catatan.';
                            }
                        }
                    } else {
                        $errors[] = 'File ini bukan catatan teks yang bisa diedit.';
                    }
                } else {
                    $errors[] = 'Catatan tidak ditemukan / bukan milik Anda.';
                    $stmt->close();
                }
            } else {
                $errors[] = 'Gagal menyiapkan query pencarian catatan.';
            }
        }

    // 5) Hapus file
    } elseif ($action === 'delete_file') {
        $fileId = (int)($_POST['file_id'] ?? 0);
        if ($fileId <= 0) {
            $errors[] = 'File tidak ditemukan.';
        } else {
            $stmt = $db->prepare("
                SELECT id, folder_id, stored_name
                FROM study_files
                WHERE id = ? AND user_id = ?
                LIMIT 1
            ");
            if ($stmt) {
                $stmt->bind_param("ii", $fileId, $userId);
                $stmt->execute();
                $res = $stmt->get_result();
                if ($res && $res->num_rows) {
                    $row      = $res->fetch_assoc();
                    $folderId = (int)$row['folder_id'];
                    $stored   = $row['stored_name'];
                    $stmt->close();

                    $stmtD = $db->prepare("DELETE FROM study_files WHERE id = ? AND user_id = ? LIMIT 1");
                    if ($stmtD) {
                        $stmtD->bind_param("ii", $fileId, $userId);
                        if ($stmtD->execute()) {
                            $fs = $uploadBaseDir . '/' . $stored;
                            if (is_file($fs)) {
                                @unlink($fs);
                            }
                            header('Location: ' . $baseUrl . '/space_belajar/file_exploler.php?folder=' . $folderId);
                            exit;
                        } else {
                            $errors[] = 'Gagal menghapus file dari database: ' . $stmtD->error;
                        }
                        $stmtD->close();
                    } else {
                        $errors[] = 'Gagal menyiapkan query hapus file.';
                    }
                } else {
                    $errors[] = 'File tidak ditemukan / bukan milik Anda.';
                    $stmt->close();
                }
            } else {
                $errors[] = 'Gagal menyiapkan query pencarian file.';
            }
        }
    }
}

// Breadcrumb (mirip path bar Windows)
$breadcrumbs = [];
$crumbId     = $currentFolderId;
while ($crumbId > 0) {
    $stmt = $db->prepare("
        SELECT id, parent_id, name
        FROM study_folders
        WHERE id = ? AND user_id = ?
        LIMIT 1
    ");
    if (!$stmt) break;
    $stmt->bind_param("ii", $crumbId, $userId);
    $stmt->execute();
    $res = $stmt->get_result();
    if (!$res || !$res->num_rows) {
        $stmt->close();
        break;
    }
    $row = $res->fetch_assoc();
    $stmt->close();
    $breadcrumbs[] = $row;
    $crumbId = (int)$row['parent_id'];
}
$breadcrumbs = array_reverse($breadcrumbs);

// Sub folder
$folders = [];
$stmt = $db->prepare("
    SELECT id, name, parent_id, created_at
    FROM study_folders
    WHERE user_id = ? AND parent_id = ?
    ORDER BY name ASC
");
if ($stmt) {
    $stmt->bind_param("ii", $userId, $currentFolderId);
    $stmt->execute();
    $res = $stmt->get_result();
    if ($res) {
        $folders = $res->fetch_all(MYSQLI_ASSOC);
    }
    $stmt->close();
}

// Files (dengan search)
$files = [];
$sqlFiles = "
    SELECT id, user_id, folder_id, title, original_name, stored_name, mime_type, size_bytes, description, created_at
    FROM study_files
    WHERE user_id = ?
      AND (
            (folder_id IS NULL AND ? = 0)
            OR folder_id = ?
          )
";
$params = [$userId, $currentFolderId, $currentFolderId];
$types  = 'iii';

if ($searchQuery !== '') {
    $sqlFiles .= " AND (title LIKE ? OR original_name LIKE ? OR description LIKE ?)";
    $like = '%' . $searchQuery . '%';
    $params[] = $like;
    $params[] = $like;
    $params[] = $like;
    $types   .= 'sss';
}
$sqlFiles .= " ORDER BY created_at DESC";

$stmt = $db->prepare($sqlFiles);
if ($stmt) {
    $bind = [];
    $bind[] = &$types;
    foreach ($params as $k => $v) {
        $params[$k] = $v;
        $bind[] = &$params[$k];
    }
    call_user_func_array([$stmt, 'bind_param'], $bind);
    $stmt->execute();
    $res = $stmt->get_result();
    if ($res) {
        $files = $res->fetch_all(MYSQLI_ASSOC);
    }
    $stmt->close();
}
